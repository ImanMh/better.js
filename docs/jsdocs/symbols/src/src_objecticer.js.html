1 /**
  2  * use Proxy API to freeze access and creation of unexisting property.
  3  * After this, if you read a unexisting property, you will get an exception, instead of the usual undefined.
  4  * After this, if you write on a unexisting property, you will get an exception, instead of a new property.
  5  * 
  6  * @param  {Object} target the object to protect
  7  * @param  {String} permission 'read' to protect only read, 'write' for write only, 'rw' for both
  8  * @return {Object} the protected object
  9  */
 10 var ObjectIcer	= function(target, permission){
 11 	console.assert(ObjectIcer.isAvailable, 'harmony Proxy not enable. try chrome://flags or node --harmony')
 12 	permission	= permission	|| 'rw'
 13 	console.assert(permission === 'rw' || permission === 'write' || permission === 'read', 'permission argument is invalid')
 14 	var checkRead	= permission === 'read'	|| permission === 'rw'
 15 	var checkWrite	= permission === 'write'|| permission === 'rw'
 16 	// use old proxy API because v8 doesnt have the new API, firefox got it tho
 17 	return Proxy.create({
 18 		get	: function(receiver, name){
 19 			if( checkRead && (name in target) === false ){
 20 				console.assert((name in target) === true, 'reading unexisting property '+name)			
 21 			}
 22 			return target[name]
 23 		},
 24 		set	: function(receiver, name, value){
 25 			if( checkWrite && (name in target) === false ){
 26 				console.assert((name in target) === true, 'setting unexisting property '+name)
 27 			}
 28 			target[name]	= value
 29 			return true
 30 		},
 31 		// without this one, i receive errors ?
 32 		has	: function (name) {
 33 			return name in target ? true : false
 34 		},
 35 	})
 36 }
 37 
 38 /**
 39  * check if the feature is available or not
 40  * @type {Boolean}
 41  */
 42 ObjectIcer.isAvailable        = typeof(Proxy) !== 'undefined' && typeof(Proxy.create) === 'function'
 43 
 44 // export the class in node.js - if running in node.js
 45 if( typeof(window) === 'undefined' )	module.exports	= ObjectIcer;
