1 /**
  2  * @fileOverview definition of PropertyAttr - based on other core libraries
  3  */
  4 
  5 var StrongTyping= StrongTyping	|| require('../strongtyping.js');
  6 var Privatize	= Privatize	|| require('../privatize.js');
  7 
  8 /**
  9  * [PropertyAttr description]
 10  * 
 11  * @param {Object} baseObject the base object
 12  * @param {String} property   the property name
 13  * @param {Object} attributes the attributes for this property
 14  */
 15 var PropertyAttr	= function(baseObject, property, attributes){
 16 	// honor .type	
 17 	if( attributes.type ){
 18 		var allowedType	= attributes.type
 19 		StrongTyping.setter(baseObject, property, allowedType)
 20 	}
 21 
 22 	// honor .private
 23 	if( attributes.private ){
 24 		Privatize.property(baseObject, property)	
 25 	}
 26 }
 27 
 28 
 29 // export the class in node.js - if running in node.js
 30 if( typeof(window) === 'undefined' )	module.exports	= PropertyAttr;
