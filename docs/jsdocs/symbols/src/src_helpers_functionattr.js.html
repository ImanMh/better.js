 33 /**
 34  * [FunctionAttr description]
 35  * 
 36  * @param {Object} baseObject the base object
 37  * @param {String} property   the property name
 38  * @param {Object} attributes the attributes for this property
 39  */
 40 var FunctionAttr	= function(originalFn, attributes){
 41 	var functionName= attributes.name	|| originalFn.name
 42 	
 43 	var privateDone	= false
 44 
 45 	return wrapFunction(originalFn, functionName, function(instance, args){
 46 		// honor .private
 47 		if( attributes.private === true && privateDone === false ){
 48 			Privatize.pushPrivateOkFn(instance, originalFn)
 49 			privateDone	= true
 50 		}
 51 
 52 		// honor .arguments - check arguments type
 53 		if( attributes.arguments !== undefined ){
 54 			var allowedTypes	= attributes.arguments
 55 			console.assert(args.length <= allowedTypes.length, 'function received '+args.length+' parameters but allows only '+allowedTypes.length+'!');
 56 			for(var i = 0; i < allowedTypes.length; i++){
 57 				var isValid	= StrongTyping.value(args[i], allowedTypes[i]);
 58 				console.assert(isValid, 'argument['+i+'] type is invalid. MUST be of type', allowedTypes[i], 'It is ===', arguments[i])
 59 			}			
 60 		}
 61 		
 62 	}, function(returnedValue, instance, args){
 63 		// honor .return - check the result type
 64 		if( attributes.return !== undefined ){
 65 			var allowedTypes= attributes.return
 66 	// console.log('blabla', arguments)
 67 			var isValid	= StrongTyping.value(returnedValue, allowedTypes)
 68 			console.assert(isValid, 'invalid type for returned value. MUST be of type', allowedTypes, 'It is ===', returnedValue)			
 69 		}
 70 	})
 71 	
 72 	return
 73 	//////////////////////////////////////////////////////////////////////////////////
 74 	//		comment								//
 75 	//////////////////////////////////////////////////////////////////////////////////
 76 	
 77 	function wrapFunction(originalFn, functionName, onBefore, onAfter){
 78 		// arguments default
 79 		onBefore= onBefore	|| function(/* ... */){}
 80 		onAfter	= onAfter	|| function(/* ... */){}
 81 		var fn	= function SuperName(){
 82 			// notify onBefore
 83 			onBefore(this, arguments)
 84 			// forward the call to original contructor
 85 			var returnedValue	= originalFn.apply(this, arguments);
 86 			// notify onAfter				
 87 			onAfter(returnedValue, this, arguments)
 88 			// actually return the value
 89 			return returnedValue;
 90 		}
 91 		// mechanism to get fn with the propername
 92 		// - see https://github.com/jeromeetienne/creatorpattern.js
 93 		var jsCode	= fn.toString().replace(/SuperName/g, functionName) 
 94 		eval('fn = '+jsCode+';')
 95 		// return the just built function
 96 		return fn
 97 	}
 98 }
